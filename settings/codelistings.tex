%define Javascript language
\lstdefinelanguage{JavaScript}{
	keywords={typeof, new, true, false, catch, function, return, null, catch, switch, var, const, let, if, in, while, do, else, case, break, object},
	keywordstyle=\color{colKeys},
	ndkeywords={class, export, boolean, throw, implements, import, this, render},
	ndkeywordstyle=\color{colNumbers}\bfseries,
	identifierstyle=\color{colIdentifier},
	sensitive=false,
	comment=[l]{//},
	morecomment=[s]{/*}{*/},
	commentstyle=\color{colGray}\ttfamily,
	stringstyle=\color{colString}\ttfamily,
	morestring=[b]',
	morestring=[b]"
}
