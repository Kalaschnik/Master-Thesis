\SetWatermarkAngle{90}
\makeatletter
\renewcommand\sc@watermark{%
  \setlength{\@tempdimb}{.072\paperwidth}% change here as you need horiz pos
  \setlength{\@tempdimc}{-.5\paperheight}%  change vertical pos
  \put(\strip@pt\@tempdimb,\strip@pt\@tempdimc){%
    \makebox(0,0){\rotatebox{\sc@wm@angle}{%
        \scalebox{\sc@wm@scale}{%
          \expandafter\expandafter\expandafter\textcolor
          \expandafter\sc@wm@colormodel\expandafter{\sc@wm@colorspecs}{%
            \fontsize{\sc@wm@fontsize}{1.2\sc@wm@fontsize}
            \selectfont
            \sc@wm@text}}}}}}
\makeatother


% defines a switch: \draftwatermarkon and \draftwatermarkoff
\makeatletter
\let\originalsc@watermark\sc@watermark
\newcommand{\draftwatermarkon}{%
  \let\sc@watermark\originalsc@watermark
}
\newcommand{\draftwatermarkoff}{%
  \let\sc@watermark\@empty
}
\makeatother
